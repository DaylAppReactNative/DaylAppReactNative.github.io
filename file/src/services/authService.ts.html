    import axios from "axios";
import {LOGIN, SALT, SIGNUP, URL} from "./constantes";
import {AuthData} from "../../types";
import sha1 from "sha1";
import Toast from "react-native-toast-message";

let id = 0

/**
 * Function executing an API call to post and save the user credentials on signup
 * @param {string} email
 * @param {string} _password
 * @param {string} firstName
 * @param {string} lastName
 */
const signUp = (email: string, _password: string, firstName: string, lastName: string): Promise<AuthData> => {
    return axios
        .post(SIGNUP, {
                firstName: firstName,
                lastName: lastName,
                email: email,
                password: sha1(_password + SALT),
                userType: "amateur",
                pseudo: "test"
            }
            , {headers: {"Access-Control-Allow-Origin": "*", "Content-Type": "application/json"}})
        .then((response) => {
            return response.data
        });
};

/**
 * Function executing an API call to verify the user credentials on log in
 * @param {string} email
 * @param {string} _password
 */
const logIn = (email: string, _password: string): Promise<AuthData> => {
    return axios
        .post(LOGIN, {
                email: email,
                password: sha1(_password + SALT),
            }
            , {headers: {"Access-Control-Allow-Origin": "*", "Content-Type": "application/json"}})
        .then((response) => {
            id = response.data.id
            return response.data
        })
        .catch(err => {
            if (err.response) {
                Toast.show({
                    type: "dayl",
                    props: {
                        title: "Error",
                        text: "Wrong password or wrong email",
                    }
                })
            }
        });
};

/**
 * API call to get the user data on reload
 */
const reloadData = () => {
    return axios
        .get(URL + `users/${id}`).then((response) => {
            return response.data
        })
}

/**
 * Function executing an API call to delete an artwork from a list
 * @param {string[]} listToDelete
 * @param {string | undefined} listId
 */
const deleteArtworks = (listToDelete: string[], listId: string | undefined) => {
    if (listId !== undefined) {
        listToDelete.forEach(paintingId => {
            axios.get(URL + "userLists/removePaintingFromList", {params: {paintingId: paintingId, listId: listId}})
                .then((response) => {
                    Toast.show({
                        type: "dayl",
                        props: {
                            title: "Success",
                            text: "Successfully removed from liked artworks",
                        }
                    })
                })
                .catch(err => {
                    if (err.response) {
                        Toast.show({
                            type: "dayl",
                            props: {
                                title: "Error",
                                text: "Could not remove from list",
                            }
                        })
                    }
                });
        })
    } else {
        listToDelete.forEach(paintingId => {
            axios.post(URL + "users/updateFavoritePainting", {
                    userId: id,
                    favoriteId: paintingId,
                    operation: "remove"
                },
                {headers: {"Access-Control-Allow-Origin": "*", "Content-Type": "application/json", "USER_PAINT": 0}}
            ).catch(err => {
                if (err.response) {
                    Toast.show({
                        type: "dayl",
                        props: {
                            title: "Error",
                            text: "Could not remove from favorite artworks",
                        }
                    })
                }
            });
        })
    }
}

/**
 * Function executing an API call to create a new list
 * @param {string} listTitle
 */
const createNewList = (listTitle: string) => {
    axios.post(URL + "userLists/save",
        {
            listName: listTitle,
            rank: 0,
            userId: id,
            id: 0,
            status: "ACTIVE",
            detail: "",
            imageUrl: ""
        }, {headers: {"Access-Control-Allow-Origin": "*", "Content-Type": "application/json"}}
    )
}

/**
 * Function executing an API call to update a list
 * @param {string} listId
 * @param {string} newName
 */
const updateList = (listId: string, newName: string) => {
    axios.get(URL + `userLists/detailByListId`, {params: {listId: listId}}).then((response) => {
        let detail = response.data.userListDTO.detail;
        axios.post(URL + "userLists/update", {
                id: listId,
                listName: newName,
                detail: detail
            },
            {headers: {"Access-Control-Allow-Origin": "*", "Content-Type": "application/json", "USER_PAINT": 0}}
        )
    });
}

/**
 * Function executing an API call to delete a list
 * @param {string} listId
 */
const deleteList = (listId: string) => {
    axios.delete(URL + `userLists/deleteById?listId=${listId}`, {
        headers: {
            "Access-Control-Allow-Origin": "*",
            "Content-Type": "application/json"
        }
    })
}

/**
 * Function executing an API call to add an artwork to a list
 * @param {string} listId
 * @param {string[]} selectedArtworks
 */
const addArtworksToList = (listId: string, selectedArtworks: string[]) => {
    axios.get(URL + `userLists/detailByListId`, {params: {listId: listId}}).then((response) => {
        const data = response.data.listPaintings.map((el) => el.id)

        selectedArtworks.forEach(paintingId => {

            if (!data.includes(paintingId)) {
                axios.get(URL + "userLists/addPaintingToList", {
                    params: {
                        listId: listId,
                        paintingId: paintingId
                    }
                }).catch(err => {
                    if (err.response) {
                        Toast.show({
                            type: "dayl",
                            props: {
                                title: "Error",
                                text: "Error adding to list",
                            }
                        })
                    }
                });
            }
            else {
                Toast.show({
                    type: "dayl",
                    props: {
                        title: "Error",
                        text: "Already in list",
                    }
                })
            }
        })

    });



}

/**
 * Functions
 */
export const authService = {
    signUp,
    logIn,
    reloadData,
    deleteArtworks,
    createNewList,
    updateList,
    deleteList,
    addArtworksToList
}
