    import {ScrollView, StyleSheet, Text, TouchableOpacity} from 'react-native';
import React, {useState} from "react";
import HeaderEditProfileComponent from "../components/HeaderComponent/HeaderEditProfileComponent";
import {useAuth} from "../contexts/Auth";
import {URL} from "../services/constantes"
import EditProfileInput from "../components/Utilities/EditProfileInput";
import colors from "../../constants/Colors";
import axios from "axios";
import Toast from "react-native-toast-message";
import BiographyInput from "../components/Utilities/BiographyInput";

/**
 * Page on which the user can edit fields of his profile (First Name, Last Name, Bio, City, Country, IconImage, BannerImage)
 *
 * <img src="https://drive.google.com/uc?export=view&id=1kE2Rf5sJ5bE5drRhAn1yba5plaIyhuoh" alt="EditProfile" height="500">
 * @constructor
 * @return EditProfile
 */
export default function EditProfile() {

    const auth = useAuth()

    const [firstName, setFirstName] = useState(auth.authData.firstName)
    const [lastName, setLastName] = useState(auth.authData.lastName)
    const [profile, setProfile] = useState(auth.authData.biography)
    const [city, setCity] = useState(auth.authData.city)
    const [country, setCountry] = useState(auth.authData.country)

    // API POST CALL TO UPDATE THE PROFILE
    const submitNewProfile = () => {
        if (firstName === "")
            Toast.show({
                type: "dayl",
                props: {
                    title: "Error",
                    text: "Please enter first name",
                }
            })
        else if (lastName === "")
            Toast.show({
                type: "dayl",
                props: {
                    title: "Error",
                    text: "Please enter last name",
                }
            })
        else {
            axios.post(URL + "users/update",
                {
                    id: auth.authData.id,
                    firstName: firstName,
                    lastName: lastName,
                    biography: profile,
                    linkedinAccount: "",
                    instragramAccount: "",
                    facebookAccount: "",
                    pinterestAccount: "",
                    twitterAccount: "",
                    city: city,
                    country: country
                }, {headers: {"Access-Control-Allow-Origin": "*", "Content-Type": "application/json"}})
                .then((response) => {
                    Toast.show({
                        type: "dayl",
                        props: {
                            title: "Success",
                            text: "Successfully updated profile",
                        }
                    })
                    auth.reloadData();
                })
                .catch(err => {
                    if (err.response) {
                        Toast.show({
                            type: "dayl",
                            props: {
                                title: "Error",
                                text: "Could not update profile, please try again later",
                            }
                        })
                    }
                });
        }
    }

    return (
        <ScrollView>
            {/*HEADER*/}
            <HeaderEditProfileComponent
                imageUri={auth.authData.backgroundImage === "" ? "https://upload.wikimedia.org/wikipedia/commons/thumb/4/49/A_black_image.jpg/640px-A_black_image.jpg" : URL + `images/contentUser/${auth.authData.id}/${auth.authData.backgroundImage}`}
                title={auth.authData.firstName + " " + auth.authData.lastName}
                avatarIconUri={auth.authData.iconImage === "" ? "https://vetref.fr/wp-content/uploads/2021/02/blank-profile-picture-973460_640.png" : URL + `images/contentUser/${auth.authData.id}/${auth.authData.iconImage}`}/>

            {/*EDITING FIELD FOR FIRST NAME*/}
            <EditProfileInput credentialsNeeded={"First name"} value={firstName} setValue={setFirstName}
                              placeholder={auth.authData.firstName}/>
            {/*EDITING FIELD FOR LAST NAME*/}
            <EditProfileInput credentialsNeeded={"Last name"} value={lastName} setValue={setLastName}
                              placeholder={auth.authData.lastName}/>
            {/*EDITING FIELD FOR BIO*/}
            <BiographyInput credentialsNeeded={"Biography"} value={profile} setValue={setProfile}/>
            {/*EDITING FIELD FOR CITY*/}
            <EditProfileInput credentialsNeeded={"City"} value={city} setValue={setCity}
                              placeholder={auth.authData.city}/>
            {/*EDITING FIELD FOR COUNTRY*/}
            <EditProfileInput credentialsNeeded={"Country"} value={country} setValue={setCountry}
                              placeholder={auth.authData.country}/>
            {/*SUBMIT BUTTON*/}
            <TouchableOpacity style={styles.saveButton} onPress={() => {submitNewProfile()}}>
                <Text style={styles.saveText}>UPDATE PROFILE</Text>
            </TouchableOpacity>
        </ScrollView>
    )
}
