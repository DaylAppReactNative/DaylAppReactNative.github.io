    //ARTWORK PAGE
import {Animated, StyleSheet, View} from 'react-native';
import React, {useEffect, useState} from "react";
import {HeaderType} from "../types";
import BackButton from "../components/Utilities/BackButton";
import OnClickImage from "../components/Utilities/OnClickImage";
import TopBarAnimatedHeader from "../components/TopBarAnimatedHeader";
import {IMG1, IMG2, LOREM, PP1, PP2} from "../services/constantes";
import TabViewComponent, {
    COLLAPSED_HEIGHT_ARTIST,
    HEADER_HEIGHT_ARTIST,
    SCROLLABLE_HEIGHT_ARITST
} from "../components/TabViewComponent";
import {TabBar} from "react-native-tab-view";
import Informations from "../components/Information";
import PaintingsFromSameArtist from "../components/ArtworkCarouselDouble/PaintingsFromSameArtist";
import SimilarArtists from "../components/SquareComponentGrid/SimilarArtists";
import PostComponent from "../components/SocialNetwork/Posts/PostComponent";
import HeaderArtistNonUser from "../components/HeaderComponent/HeaderArtistNonUserComponent";
import FollowingButtonComponent from "../components/Utilities/FollowingButtonComponent";
import colors from "../constants/Colors";
import {PostType, RoleType} from "../../types";


const postS = <PostComponent type={PostType.Social} image={IMG1} text={LOREM} avatarUri={PP1}
                             userName={"Pablo Picasso"} userId={"1"} verified={true} repost={true} threedots={false}
                             date={"6hrs ago"}/>

const postP = <PostComponent type={PostType.Painting} artworkName={"Nuit étoilée"} image={IMG2} avatarUri={PP2}
                             userName={"Vincent Van Gogh"}
                             userId={"2"} verified={true} repost={true} threedots={false} date={"6hrs ago"}/>

/**
 * @ignore
 * @param route
 * @constructor
 */
export default function OldArtistPage({route}: any) {

    const {
        imageUri,
        information,
        artistName,
        artistId,
    }: {
        imageUri: string,
        information: string,
        artistName: string,
        artistId: string,
    } = route.params


    const scrollY = (new Animated.Value(0));

    useEffect(() => {
        scrollY.setValue(0)
    }, [artistId])

    const [onClickImage, setOnClickImage] = useState(false)

    let [pointerEventTopBar, setPointerEvenTopBar] = useState<any>("none")

    //ANIMATION
    const translateY = scrollY.interpolate({
        inputRange: [0, SCROLLABLE_HEIGHT_ARITST],
        outputRange: [0, -SCROLLABLE_HEIGHT_ARITST],
        extrapolate: 'clamp',
    });

    let translateY2 = scrollY.interpolate({
        inputRange: [-100, 0], outputRange: [150, 0], extrapolateLeft: "extend", extrapolateRight: "clamp"
    })


    const firstRoute = () => (
        <Animated.ScrollView scrollEventThrottle={1}
                             onScroll={Animated.event(
                                 [{nativeEvent: {contentOffset: {y: scrollY}}}],
                                 {useNativeDriver: true,}
                             )}

                             contentContainerStyle={{paddingTop: HEADER_HEIGHT_ARTIST + COLLAPSED_HEIGHT_ARTIST}}>
            {/* SUMMARY ON ARTIST */}
            <Informations title={"Biography"} text={information}/>
            {/*    MORE PAINTINGS FROM SAME ARTIST*/}
            <PaintingsFromSameArtist artistId={artistId}/>
            {/*    SIMILAR ARTISTS*/}
            <SimilarArtists artistId={artistId}/>
        </Animated.ScrollView>)


    const secondRoute = () => (
        //    ANALYSIS
        <Animated.ScrollView style={{backgroundColor: 'black', marginTop: 10}} scrollEventThrottle={1}
                             onScroll={Animated.event(
                                 [{nativeEvent: {contentOffset: {y: scrollY}}}],
                                 {useNativeDriver: true}
                             )}
                             contentContainerStyle={{paddingTop: HEADER_HEIGHT_ARTIST + COLLAPSED_HEIGHT_ARTIST}}>
            <PostComponent type={PostType.Analysis} avatarUri={PP1} userName={"Pablo Picasso"} verified={true} role={RoleType.EXPERT}
                           text={LOREM} repost={false} threedots={false} userId={"1"}/>
            <PostComponent type={PostType.Analysis} avatarUri={PP1} userName={"Pablo Picasso"} verified={true} role={RoleType.EXPERT}
                           text={LOREM} repost={false} threedots={false} userId={"1"}/>
            <PostComponent type={PostType.Analysis} avatarUri={PP1} userName={"Pablo Picasso"} verified={true} role={RoleType.EXPERT}
                           text={LOREM} repost={false} threedots={false} userId={"1"}/>
        </Animated.ScrollView>
    )


    const thirdRoute = () => (
        //    ACTIVITY
        <Animated.ScrollView style={{backgroundColor: 'black', marginTop: 10}} scrollEventThrottle={1}
                             onScroll={Animated.event(
                                 [{nativeEvent: {contentOffset: {y: scrollY}}}],
                                 {useNativeDriver: true,}
                             )}
                             contentContainerStyle={{paddingTop: HEADER_HEIGHT_ARTIST + COLLAPSED_HEIGHT_ARTIST}}>
            <PostComponent type={PostType.Social} image={IMG1} text={LOREM} avatarUri={PP1}
                           userName={"Pablo Picasso"} userId={"1"} verified={true} repost={false} threedots={false}
                           date={"6hrs ago"}/>
            <PostComponent type={PostType.Painting} artworkName={"Nuit étoilée"} image={IMG2} avatarUri={PP2}
                           userName={"Vincent Van Gogh"}
                           userId={"2"} verified={true} repost={false} threedots={false} date={"6hrs ago"}/>
            <PostComponent type={PostType.RepostSocial} image={IMG1} text={LOREM} avatarUri={PP1}
                           userName={"Pablo Picasso"} userId={"3"} verified={true} repost={false} threedots={false}
                           date={"6hrs ago"} post={postS}/>
            <PostComponent type={PostType.RepostPainting} image={IMG1} text={LOREM} avatarUri={PP1}
                           userName={"Pablo Picasso"} userId={"4"} verified={true} repost={false} threedots={false}
                           date={"6hrs ago"} post={postP}/>
        </Animated.ScrollView>
    )

    const header = (props: any) => {


        return (
            <Animated.View style={[styles.header,]}>
                <BackButton/>

                {/*DEACTIVATE TOP BAR AFTER SCROLL*/}
                <View style={[styles.topBar]} pointerEvents={pointerEventTopBar}>
                    <TopBarAnimatedHeader title={artistName} scrollY={scrollY} type={HeaderType.NonUser}/>
                </View>

                {/*WHEN IMAGE IS SCALED UP MOVE COMPONENTS DOWN*/}
                <Animated.View style={[{transform: [{translateY}]}]}>
                    <HeaderArtistNonUser imageUri={imageUri} artistName={artistName} scrollY={scrollY}

                                         onClickImage={onClickImage} setOnClickImage={setOnClickImage}/>
                    <View>
                        <FollowingButtonComponent artistId={artistId}/>
                    </View>
                    <TabBar {...props} indicatorStyle={{backgroundColor: colors.daylRed.text}}
                            style={[styles.tabbar, {transform: [{translateY: translateY2}]}]}/>
                </Animated.View>
            </Animated.View>)
    }

    return (
        <View style={{flex: 1}}>

            {/*ON CLICK SHOW IMAGE FULL SIZE*/}
            <OnClickImage imageUri={imageUri} title={artistName} onClickImage={onClickImage}
                          setOnClickImage={setOnClickImage} artistName={artistName}/>

            {/*RENDER HEADER + TAB VIEWS*/}
            <TabViewComponent header={header} firstRoute={firstRoute} secondRoute={secondRoute}
                              thirdRoute={thirdRoute}/>

        </View>
    )
}

const styles = StyleSheet.create({
    topBar: {
        zIndex: 4
    },
    header: {
        position: 'absolute',
        top: 0,
        left: 0,
        right: 0,
        zIndex: 1,
    },
    tabbar: {
        backgroundColor: 'black',
        elevation: 0,
        shadowOpacity: 0
    },

});
