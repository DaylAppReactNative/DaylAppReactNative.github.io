    import {Animated, Button, Platform, SafeAreaView, ScrollView, StyleSheet, Text, TouchableOpacity, View} from 'react-native';
import React, {useEffect, useRef, useState} from "react";
import colors from "../../constants/Colors";
import TopBarAnimatedHeader from "../components/TopBarAnimatedHeader";
import {Artwork, HeaderType, ListActions, PageTypeList} from "../../types";
import axios from "axios";
import {useAuth} from "../contexts/Auth";
import {URL} from "../services/constantes"
import HeaderListComponent from "../components/HeaderComponent/HeaderListComponent";
import SmallerSearchBar from "../components/Utilities/SearchBar/SmallerSearchBar";
import layout from "../../constants/Layout";
import EditListModal from "../components/Modals/EditListModal";
import ToolBarListPage from "../components/Utilities/ToolBarListPage";
import MyListsComponent from "../components/SquareComponentGrid/MyLists";
import ArtworkGrid from "../components/ArtworkGrid";
import {useInteractions} from "../contexts/UserInteractions";
import Modal from "react-native-modal";
import {Entypo} from "@expo/vector-icons";

/**
 * Page used to present the content of a list
 *
 * <img src="https://drive.google.com/uc?export=view&id=1vpaQ6Y68QHQyUf6d-oqM_pSmPF1FbQXU" alt="ArtworkPage" height="500">
 * @param {Object} route
 * @param {string | undefined} route.id - ID of the list
 * @param {string} route.name - Name of the list
 * @param {Artwork[]} route.data - Array of artwork(s) belonging to the list
 * @constructor
 * @return ListContentPage
 */
export default function ListContentPage({route}: any) {

    const {
        id,
        name,
        data
    }: {
        id: string | undefined,
        name: string,
        data?: Artwork[]
    } = route.params

    const userInt = useInteractions()

    const scrollY = useRef(new Animated.Value(0)).current;

    let [pointerEventTopBar, setPointerEvenTopBar] = useState<any>("none")

    const [painting, setPainting] = useState<[Artwork] | []>([]);

    const [clicked, setClicked] = useState(false)

    const [searchPhrase, setSearchPhrase] = useState("")

    const [newData, setNewData] = useState(data)

    const [modalVisible, setModalVisible] = useState(false)

    const [addToListModal, setAddToListModal] = useState(false)

    const auth = useAuth()

    useEffect(() => {
        const temp = painting.filter(item => item.artworkName.toLowerCase().includes(searchPhrase.toLowerCase()) || item.artistName.toLowerCase().includes(searchPhrase.toLowerCase()))
        setNewData(temp)
    }, [searchPhrase])

    useEffect(() => {
        userInt.setEdit(false)
        if (data)
            setNewData(data)
    }, [])

    if (!data) {
        if (id === undefined) {
            useEffect(() => {

                if (auth.authData.paintingLiked) {
                    setPainting([])
                    for (let i = 0; i < auth.authData.paintingLiked.length; i++)
                        axios.get(URL + `paintingEntity/${auth.authData.paintingLiked[i]}`).then((response) => {

                            let data = response.data;
                            let parsedData: Artwork = {
                                id: data.id,
                                imageUri: URL + `images/content/mini/${data.artist_idtf_intern}/${data.generatedId}`,
                                artistName: data.artist_name_tot,
                                artworkName: data.aw_title_Eng,
                                artistId: data.artist_idtf_intern
                            }
                            // @ts-ignore
                            setPainting(painting => [...painting, parsedData]);
                            setNewData(painting)
                        });
                }
            }, [auth.authData.paintingLiked]);

        } else {
            useEffect(() => {
                setPainting([])
                axios.get(URL + `userLists/detailByListId`, {params: {listId: id}}).then((response) => {
                    let data = response.data.listPaintings;
                    const parsedData = data.map((el: any) => {
                        return {
                            id: el.id,
                            imageUri: URL + `images/content/mini/${el.artist_idtf_intern}/${el.generatedId}`,
                            artistName: el.artist_name_tot,
                            artworkName: el.aw_title_Eng,
                            artistId: el.artist_idtf_intern
                        }
                    })
                    setPainting(parsedData)
                });
            }, [id, auth.refreshKey]);
        }
    } else {
        useEffect(() => {
            setPainting(data)
        }, [])
    }


    return (
        <View style={styles.container}>
            {/*ANIMATED TOP BAR THAT APPEARS WHEN SCROLL DOWN*/}
            <View style={[styles.topBar]} pointerEvents={pointerEventTopBar}>
                <TopBarAnimatedHeader title={name} type={HeaderType.List} scrollY={scrollY}/>
            </View>

            <Modal isVisible={addToListModal}
                   animationIn={"fadeIn"}
                   animationOut={"fadeOut"}
                   backdropOpacity={0}
                   onBackButtonPress={() => setAddToListModal(false)}
                   statusBarTranslucent={true}
                   //swipeDirection={["down"]}
                   //propagateSwipe={true}
                   //onSwipeComplete={() => setAddToListModal(false)}
                   style={styles.modal}>
                <View style={styles.scrollViewModalContainer}>

                    <ScrollView bounces={false} scrollEventThrottle={16}>

                        <View style={styles.headerModalContainer}>

                            <TouchableOpacity style={styles.backButton} onPress={() => setAddToListModal(false)}>
                                <Entypo name={"chevron-left"} size={35} color={"white"}/>
                            </TouchableOpacity>
                            <Text style={styles.titleModal}>Add to another list</Text>
                        </View>

                         <MyListsComponent action={ListActions.add}/>

                    </ScrollView>

                </View>
            </Modal>

            {/*ANIMATED SCROLL VIEW*/}
            <Animated.ScrollView alwaysBounceVertical={true} scrollEventThrottle={16}
                                 showsVerticalScrollIndicator={false}
                                 onScroll={Animated.event([{nativeEvent: {contentOffset: {y: scrollY}}}], {
                                     useNativeDriver: true,
                                     listener: (event: any) => setPointerEvenTopBar(event.nativeEvent.contentOffset.y > 90 ? "auto" : "none")
                                 })}>

                {/*ANIMATED HEADER*/}
                <HeaderListComponent listName={name} scrollY={scrollY}
                                     fourImages={data ? searchPhrase === "" ? painting : newData : painting}/>

                {/*Search Bar*/}
                {id && <SmallerSearchBar clicked={clicked} searchPhrase={searchPhrase} setSearchPhrase={setSearchPhrase}
                                         setClicked={setClicked}/>}

                {/* ARTWORK GRID */}
                <ArtworkGrid data={searchPhrase === "" ? painting : newData}/>

                {<ToolBarListPage type={!data ? PageTypeList.MyList : PageTypeList.OtherList} listId={id}
                                  modalVisible={modalVisible} setModalVisible={setModalVisible}
                                  addToListModal={addToListModal} setAddToListModal={setAddToListModal}/>}

                {/*EDIT LIST MODAL*/}
                {id && <EditListModal listId={id} modalVisible={modalVisible} setModalVisible={setModalVisible}/>}

            </Animated.ScrollView>
        </View>
    )
}
