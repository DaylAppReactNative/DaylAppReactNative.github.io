    src/screens/MatchingByColorPage.tsx

    import {Platform, SafeAreaView, ScrollView, StyleSheet, TouchableOpacity} from 'react-native';
import MatchingByColorGrid from "../components/ArtworkGrid/MatchingByColorGrid";
import {MaterialIcons} from "@expo/vector-icons";
import layout from "../../constants/Layout";
import ColorPickerModal from "../components/Modals/ColorPickerModal";
import {useState} from "react";

/**
 * Page displaying artworks when the user starts a matching by color
 * @param {Object} route
 * @param {string} color - Color picked by the user
 * @constructor
 * @return MatchingByColorPage
 */
export default function MatchingByColorPage({route}: any) {

    let {
        color,
    }: {
        color: string,
    } = route.params

    const [isModalVisible, setModalVisible] = useState(false)
    const [newColor, setColor] = useState(color)

    return (
        <SafeAreaView style={styles.container}>
            {/*COLOR PICKER MODAL*/}
            <ColorPickerModal isModalVisible={isModalVisible} setModalVisible={setModalVisible} color={newColor}
                              setColor={setColor}/>
            {/*COLOR ICON - TO CHOOSE ANOTHER COLOR*/}
            <TouchableOpacity style={[{backgroundColor: `#${color}`}, styles.colorPickerButton]}
                              onPress={() => setModalVisible(true)}>
                <MaterialIcons name={"colorize"} size={30} color={"white"}/>
            </TouchableOpacity>
            {/*RESULT MATCHING ARTWORK GRID*/}
            <ScrollView showsVerticalScrollIndicator={false}>
                <MatchingByColorGrid color={color}/>
            </ScrollView>
        </SafeAreaView>
    )
}

const styles = StyleSheet.create({
    container: {
        paddingTop: Platform.OS === 'android' ? 25 : 0,
    },
    colorPickerButton: {
        justifyContent: "center",
        alignItems: "center",
        position: "absolute",
        bottom: 30,
        right: 30,
        borderRadius: 30,
        zIndex: 3,
        width: layout.window.width / 7,
        height: layout.window.width / 7
    }
});
