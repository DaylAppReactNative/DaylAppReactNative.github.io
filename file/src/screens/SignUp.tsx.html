    import {Dimensions, Platform, SafeAreaView, ScrollView, StyleSheet, Text, TouchableOpacity, View} from 'react-native';
import {LinearGradient} from "expo-linear-gradient";
import React, {useState} from "react";
import colors from "../../constants/Colors";
import {Entypo} from "@expo/vector-icons";
import CredentialsInputComponent from "../components/Utilities/CredentialsInput";
import {useAuth} from "../contexts/Auth";
import {useNavigation} from "@react-navigation/native";
import Toast from "react-native-toast-message";
import DropDownPicker from "react-native-dropdown-picker"

/**
 * Page used for signing up to the app
 *
 * <img src="https://drive.google.com/uc?export=view&id=1sc4hor1c9Kw9bqdoXmzvuKm8H9I1W2XE" alt="ArtworkPage" height="500">
 * @constructor
 * @return SignUpScreen
 */
export default function SignUpScreen() {

    const [firstName, setFirstName] = useState("")
    const [lastName, setLastName] = useState("")

    const [email, setEmail] = useState("")
    const [password, setPassword] = useState("")
    const [confirm, setConfirm] = useState("")

    const [profileType, setProfileType] = useState("")
    const [profileTypeItems, setProfileTypeItems] = useState([{label: "Amateur", value: "AMATEUR"},])
    const [openProfileType, setOpenProfileType] = useState(false)

    const auth = useAuth()

    // WHEN SIGNUP BUTTON IS PRESSED
    const onPressSignUp = () => {
        if (password != confirm) {
            Toast.show({
                type: "dayl", props: {
                    title: "Error", text: "Passwords do not match",
                }
            })
        }
        if (firstName === "" || lastName === "" || email === "" || password === "") {
            Toast.show({
                type: "dayl", props: {
                    title: "Error", text: "Please fill all fields",
                }
            })
        } else {
            const signup = auth.signUp(email, password, firstName, lastName)
        }
    }

    const navigation = useNavigation()

    return (
        <ScrollView>
            {/*DESIGN : GRADIENT*/}
            <LinearGradient colors={['black', colors.daylRed.text]} start={{x: 0, y: 1,}} end={{x: 0, y: 0}}
                            style={[styles.linearGradient]}/>

            <SafeAreaView style={styles.safeview}>
                <View style={styles.headerSignUp}>
                    {/*BACK BUTTON*/}
                    <TouchableOpacity style={styles.backButton} onPress={() => navigation.goBack()}>
                        <Entypo name={"chevron-left"} size={35} color={"white"}/>
                    </TouchableOpacity>

                    <Text style={styles.signUp}>Sign up</Text>

                    <View style={styles.emptyViewRight}></View>
                </View>
            </SafeAreaView>

            <View>
                {/*INPUT FIELD TO ENTER FIRST NAME OF USER*/}
                <CredentialsInputComponent credentialsNeeded={"First name"} value={firstName} setValue={setFirstName}/>
                {/*INPUT FIELD TO ENTER LAST NAME OF USER*/}
                <CredentialsInputComponent credentialsNeeded={"Last name"} value={lastName} setValue={setLastName}/>
                {/*<CredentialsInputComponent credentialsNeeded={"Type of profile"} value={type} setValue={setType}/>*/}
                {/*DROPDOWN TO CHOOSE ACCOUNT TYPE*/}
                <Text style={styles.accountTypeText}>Account type</Text>
                <DropDownPicker
                    placeholder={"Account type"}
                    open={openProfileType}
                    value={profileType}
                    items={profileTypeItems}
                    setOpen={setOpenProfileType}
                    setValue={setProfileType}
                    setItems={setProfileTypeItems}
                    containerStyle={styles.containerPicker}
                    style={styles.picker}
                />
                {/*INPUT FIELD TO ENTER EMAIL OF USER*/}
                <CredentialsInputComponent credentialsNeeded={"Email"} value={email} setValue={setEmail}/>
                {/*INPUT FIELD TO ENTER PASSWORD OF USER*/}
                <CredentialsInputComponent credentialsNeeded={"Password"} value={password} setValue={setPassword}/>
                {/*INPUT FIELD TO ENTER PASSWORD OF USER*/}
                <CredentialsInputComponent credentialsNeeded={"Confirm password"} value={confirm}
                                           setValue={setConfirm}/>
            </View>

            {/*SIGNUP BUTTON*/}
            <TouchableOpacity style={styles.buttonLogIn} onPress={() => onPressSignUp()}>
                <Text style={styles.buttonTextLogIn}>Sign up</Text>
            </TouchableOpacity>
        </ScrollView>
    )
}
