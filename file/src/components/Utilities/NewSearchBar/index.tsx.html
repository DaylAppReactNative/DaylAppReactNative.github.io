    src/components/Utilities/NewSearchBar/index.tsx

    import React, {useState} from "react";
import {Keyboard, TextInput, TouchableOpacity, View} from "react-native";
import {styles} from "./styles";
import {Entypo, Feather, Ionicons} from "@expo/vector-icons";
import FilterModal from "../../FilterModal";
import ModalMatchingImage from "../../Modals/MatchingImage";

// PROPS
export type SearchBarProps = {
    clicked: boolean,
    searchPhrase: string,
    setSearchPhrase: any,
    setClicked: any,
    filter: {
        styleFilter: string,
        genreFilter: string,
        nationalityFilter: string
    },
    setFilter(filter: {
        styleFilter: string,
        genreFilter: string,
        nationalityFilter: string
    }): void
}

/**
 * Component displaying a searchBar used in {@link ExplorePage}
 *
 * <img src="https://drive.google.com/uc?export=view&id=10adsPmFxl7T0smH3oiT75BiJSEAvTChV" alt="ArtworkPage" height="100">
 * @param {SearchBarProps} props
 * @param {boolean} props.clicked - Boolean checking if the searchbar is clicked
 * @param {string} props.searchPhrase - SearchPhrase text in the searchBar
 * @param {any} props.setSearchPhrase - Function to set the searchPhrase
 * @param {any} props.setClicked - Function to set {props.clicked}
 * @param {Object} props.filter - Object containing the filters
 * @param {string} props.filter.styleFilter - Style filter
 * @param {string} props.filter.genreFilter - Genre filter
 * @param {string} props.filter.nationalityFilter - Nationality filter
 * @param {Function} props.filter - Function to set the filters
 * @param {string} props.setFilter.styleFilter - Set style filter
 * @param {string} props.setFilter.genreFilter - Set genre filter
 * @param {string} props.setFilter.nationalityFilter - Set nationality filter
 * @constructor
 * @return NewSearchBar
 */
export default function NewSearchBar(props: SearchBarProps) {

    const [showFilter, setShowFilter] = useState<boolean>(false)

    return (
        <View style={{
            flexDirection: "row",
            alignSelf: "center",
            width: '100%',
            alignItems: "center",
            justifyContent: "center"
        }}>
            {(props.clicked || props.searchPhrase != "" || props.filter.styleFilter !== "" || props.filter.genreFilter !== "" || props.filter.nationalityFilter !== "") &&
                <TouchableOpacity onPress={() => {
                    Keyboard.dismiss();
                    props.setClicked(false);
                    props.setFilter({styleFilter: "", genreFilter: "", nationalityFilter: ""});
                    props.setSearchPhrase("");
                }}><Entypo name={"chevron-left"} color={"white"} size={35}/></TouchableOpacity>}
            <View
                style={props.clicked || props.searchPhrase != "" || props.filter.styleFilter !== "" || props.filter.genreFilter !== "" || props.filter.nationalityFilter !== "" ? styles.container__Clicked : styles.container}>

                <Feather name="search" color="black" style={styles.searchIcon}/>
                <TextInput style={styles.textInput} placeholder="Artworks, artists, keywords"
                           placeholderTextColor={"black"} value={props.searchPhrase}
                           onChangeText={props.setSearchPhrase} onFocus={() => {
                    props.setClicked(true)
                }}>
                </TextInput>
                <FilterModal filter={props.filter} setFilter={props.setFilter} isModalVisible={showFilter}
                             setModalVisible={setShowFilter}/>
                <View style={styles.rightContainer}>
                    <ModalMatchingImage/>
                    <Ionicons name={"filter-outline"} size={25} style={styles.icon}
                              onPress={() => setShowFilter(!showFilter)}/>
                </View>
            </View>
        </View>
    )
}
