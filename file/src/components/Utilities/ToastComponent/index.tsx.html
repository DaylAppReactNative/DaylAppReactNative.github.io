    import React from "react";
import {StyleSheet, Text, TouchableOpacity, View} from "react-native";
import Toast from "react-native-toast-message";

// COMPONENT FOR SHOWING TOAST ON TOP OF SCREEN

export type ToastComponentProps = {
    title: string; // first line
    text: string; // second line
    type: string; // success, error, warning, dayl
}

// 100% customisable toast
export const toastConfig = {
    dayl: ({props}: any) => (
        <View style={{
            width: '95%', backgroundColor: "#292929", borderRadius: 10,
            borderLeftColor: "white", borderLeftWidth: 10, paddingHorizontal: 15, paddingVertical: 15
        }}>
            <Text style={{fontSize: 15, color: "white"}}>{props.text}</Text>
        </View>
    )
}

const ToastComponent = (props: ToastComponentProps) => {

    // FUNCTION SPAWNING TOAST ONPRESS
    const showToast = () => {
        Toast.show({
            type: props.type,
            props: {
                title: props.title,
                text: props.text,
            }
        });
    }

    return (
        <TouchableOpacity style={styles.button} onPress={() => showToast()}><Text style={styles.buttonText}>TOAST TEST</Text></TouchableOpacity>
    )
}

const styles = StyleSheet.create({
    button: {
        width: "30%",
        height: 40,
        alignItems: "center",
        backgroundColor: "gray",
        alignSelf: "center",
        justifyContent: "center",
        borderRadius: 10,
        margin: 10
    },
    buttonText: {
        fontWeight: "bold",
        fontSize: 15,
        color: "white",
        textAlign: "center",
    },
})

export default ToastComponent
