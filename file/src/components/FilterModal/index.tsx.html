    import React, {useCallback, useEffect, useState} from 'react';
import {Dimensions, Text, TouchableOpacity, View} from 'react-native';
import styles from "./styles";
import Modal from "react-native-modal";
import DropDownPicker from "react-native-dropdown-picker"
import {Entypo, Feather, MaterialIcons} from "@expo/vector-icons";
import {URL} from "../../services/constantes"
import axios from "axios";

// REWORKED ARROW UP ICON FOR DROPDOWNS
const ArrowUpIcon = () => {
    return <MaterialIcons name={"keyboard-arrow-up"} size={30} color={"white"}/>
}

// REWORKED ARROW DOWN ICON FOR DROPDOWNS
const ArrowDownIcon = () => {
    return <MaterialIcons name={"keyboard-arrow-down"} size={30} color={"white"}/>
}

// REWORKED TICK ICON FOR DROPDOWNS
const TickIcon = () => {
    return <Feather name={"check"} size={20} color={"white"}/>
}

// PROPS
export type FilterModalProps = {
    filter: Object,
    setFilter(filter: any): void,
    isModalVisible: boolean,
    setModalVisible(isModalVisible: boolean): void
}

/**
 * Component used to filter search in the {@link ExplorePage}
 *
 * <img src="https://drive.google.com/uc?export=view&id=1iU5fIyy5IP-DnEbLu2Vi0PfGTOxgjVa7" alt="ArtworkPage" height="500">
 * @param {FilterModalProps} props
 * @param {Object} props.filter - Filters chosen by the user
 * @param {void} props.setFilter(filter: any) - Function for setting new filters on change
 * @param {boolean} props.isModalVisible - Modal visibility
 * @param {void} props.setModalVisible(isModalVisible: boolean) - Function for toggling modal visibility
 * @constructor
 * @return FilterModal
 */
export default function FilterModal(props: FilterModalProps) {

    // FOR OPENING/CLOSING A DROPDOWN
    // const [artMovementOpen, setArtMovementOpen] = useState(false);
    // const [colorOpen, setColorOpen] = useState(false);
    const [styleOpen, setStyleOpen] = useState(false);
    const [genreOpen, setGenreOpen] = useState(false);
    const [nationalityOpen, setNationalityOpen] = useState(false);

    // FOR CLOSING THE OTHER DROPDOWNS
    // const onArtMovementOpen = useCallback(() => {
    //     setColorOpen(false)
    //     setStyleOpen(false)
    //     setGenreOpen(false)
    //     setNationalityOpen(false)
    // }, [])
    // const onColorOpen = useCallback(() => {
    //     setArtMovementOpen(false)
    //     setStyleOpen(false)
    //     setGenreOpen(false)
    //     setNationalityOpen(false)
    // }, [])
    const onStyleOpen = useCallback(() => {
        // setArtMovementOpen(false)
        // setColorOpen(false)
        setGenreOpen(false)
        setNationalityOpen(false)
    }, [])
    const onGenreOpen = useCallback(() => {
        // setArtMovementOpen(false)
        // setColorOpen(false)
        setStyleOpen(false)
        setNationalityOpen(false)
    }, [])
    const onNationalityOpen = useCallback(() => {
        // setArtMovementOpen(false)
        // setColorOpen(false)
        setStyleOpen(false)
        setGenreOpen(false)
    }, [])

    // WHICH FILTERS THE USER CHOSE FOR EACH DROPDOWN
    // const [colorValues, setColorValues] = useState([]);
    const [styleValue, setStyleValue] = useState("");
    const [genreValue, setGenreValue] = useState("");
    const [nationalityValue, setNationalityValue] = useState("");

    useEffect(() => {
        // API CALL TO GET ALL FILTERS POSSIBLE FOR STYLES
        axios.get(URL + `paintingFront/listStyles`).then((response) => {
            // PARSING DATA
            let data = response.data.map((item: { weight: number, code: string, name: string }) => {
                return {
                    label: item.name,
                    value: item.code
                }
            })
            data.shift() // removes first empty element
            setStyleItems(data)
        });

        // API CALL TO GET ALL FILTERS POSSIBLE FOR GENRES
        axios.get(URL + `paintingFront/listGenres`).then((response) => {
            // PARSING DATA
            let data = response.data.map((item: { weight: number, code: string, name: string }) => {
                return {
                    label: item.name,
                    value: item.code
                }
            })
            data.shift() // removes first empty element
            setGenreItems(data)
        });

        // API CALL TO GET ALL FILTERS POSSIBLE FOR ORIGINS
        axios.get(URL + `paintingFront/listOrigins`).then((response) => {
            // PARSING DATA
            let data = response.data.map((item: { weight: number, code: string, name: string }) => {
                return {
                    label: item.name,
                    value: item.code
                }
            })
            data.shift() // removes first empty element
            setNationalityItems(data)
        });
    }, [])

    // LIST OF FILTERS POSSIBLE TO CHOOSE
    // const [artMovementItems, setArtMovementItems] = useState([
    //     {label: 'Symbolism', value: 'symbolism'},
    //     {label: 'Renaissance', value: 'renaissance'},
    //     {label: 'Impressionism', value: 'impressionism'},
    //     {label: 'Pointillism', value: 'pointillism'},
    // ]);
    // const [colorItems, setColorItems] = useState([
    //     {label: 'Red', value: 'red'},
    //     {label: 'Green', value: 'green'},
    // ]);
    const [styleItems, setStyleItems] = useState([
        {label: '', value: ''}
    ]);
    const [genreItems, setGenreItems] = useState([
        {label: '', value: ''}
    ]);
    const [nationalityItems, setNationalityItems] = useState([
        {label: '', value: ''}
    ]);

    const toggleModal = () => {
        props.setModalVisible(!props.isModalVisible);
    };

    // ON CLICK BUTTON FILTER => SEND FILTER DATA
    const toggleFilter = () => {
        props.setModalVisible(false)
        props.setFilter({
            // artMovementFilter: Object.values(props.artMovementValues),
            // colorFilter: Object.values(colorValues),
            styleFilter: styleValue,
            genreFilter: genreValue,
            nationalityFilter: nationalityValue
        })
    }

    // CLEAR ALL THE FILTERS VALUES
    const clearFilter = () => {
        props.setModalVisible(false)
        props.setFilter({
            styleFilter: "",
            genreFilter: "",
            nationalityFilter: ""
        })
    }

    const deviceWidth = Dimensions.get("window").width;
    const deviceHeight = Dimensions.get("window").height

    return (
        <Modal isVisible={props.isModalVisible}
               animationIn={"fadeIn"}
               animationOut={"fadeOut"}
               deviceHeight={deviceHeight}
               deviceWidth={deviceWidth}
               statusBarTranslucent={true}
               backdropOpacity={0.8}
               style={styles.containerModal}
        >
            {/*BACK BUTTON*/}
            <TouchableOpacity style={styles.backButton} onPress={() => toggleModal()}>
                <Entypo name={"chevron-left"} size={30} color={"white"}/>
            </TouchableOpacity>

            {/*TITLE*/}
            <Text style={styles.title}>Filters</Text>

            {/*DROPDOWNS AND BUTTON CONTAINER*/}
            <View style={[styles.container, {flex: 1, marginTop: 100, marginBottom: 100}]}>

                {/*DROPDOWNS CONTAINER*/}
                <View style={styles.pickersContainer}>
                    {/*<DropDownPicker
                            placeholder={"Art Movement"}
                            min={0}
                            max={5}
                            multiple={true}
                            open={artMovementOpen}
                            value={props.artMovementValues}
                            items={artMovementItems}
                            setOpen={setArtMovementOpen}
                            onOpen={onArtMovementOpen}
                            setValue={() => props.setArtMovementValues}
                            setItems={setArtMovementItems}
                            zIndex={5000}
                            zIndexInverse={1000}
                            containerStyle={styles.picker}
                            style={styles.dropdownInside}
                            labelStyle={styles.labelStyle}
                            placeholderStyle={styles.dropdownPlaceholder}
                            dropDownContainerStyle={styles.dropDownContainerStyle}
                            listItemLabelStyle={styles.listItemLabelStyle}
                            ArrowDownIconComponent={() => <ArrowDownIcon/>}
                            ArrowUpIconComponent={() => <ArrowUpIcon/>}
                            TickIconComponent={() => <TickIcon/>}
                        />*/}
                    {/*<DropDownPicker
                            placeholder={"Color"}
                            min={0}
                            max={5}
                            multiple={true}
                            open={colorOpen}
                            value={colorValues}
                            items={colorItems}
                            setOpen={setColorOpen}
                            onOpen={onColorOpen}
                            setValue={setColorValues}
                            setItems={setColorItems}
                            zIndex={4000}
                            zIndexInverse={2000}
                            containerStyle={styles.picker}
                            style={styles.dropdownInside}
                            labelStyle={styles.labelStyle}
                            placeholderStyle={styles.dropdownPlaceholder}
                            dropDownContainerStyle={styles.dropDownContainerStyle}
                            listItemLabelStyle={styles.listItemLabelStyle}
                            ArrowDownIconComponent={() => <ArrowDownIcon/>}
                            ArrowUpIconComponent={() => <ArrowUpIcon/>}
                            TickIconComponent={() => <TickIcon/>}
                        />*/}
                    {/*STYLE FILTER DROPDOWN*/}
                    <DropDownPicker
                        placeholder={"Style"}
                        min={0}
                        max={5}
                        open={styleOpen}
                        value={styleValue}
                        items={styleItems}
                        setOpen={setStyleOpen}
                        onOpen={onStyleOpen}
                        setValue={setStyleValue}
                        setItems={setStyleItems}
                        zIndex={3000}
                        zIndexInverse={1000}
                        containerStyle={styles.picker}
                        style={styles.dropdownInside}
                        labelStyle={styles.labelStyle}
                        placeholderStyle={styles.dropdownPlaceholder}
                        dropDownContainerStyle={styles.dropDownContainerStyle}
                        listItemLabelStyle={styles.listItemLabelStyle}
                        ArrowDownIconComponent={() => <ArrowDownIcon/>}
                        ArrowUpIconComponent={() => <ArrowUpIcon/>}
                        TickIconComponent={() => <TickIcon/>}
                    />

                    {/*GENRE FILTER DROPDOWN*/}
                    <DropDownPicker
                        placeholder={"Genre"}

                        open={genreOpen}
                        value={genreValue}
                        items={genreItems}
                        setOpen={setGenreOpen}
                        onOpen={onGenreOpen}
                        setValue={setGenreValue}
                        setItems={setGenreItems}
                        zIndex={3000}
                        zIndexInverse={1000}
                        containerStyle={styleOpen ? {marginTop: 200} : styles.picker}
                        style={styles.dropdownInside}
                        labelStyle={styles.labelStyle}
                        placeholderStyle={styles.dropdownPlaceholder}
                        dropDownContainerStyle={styles.dropDownContainerStyle}
                        listItemLabelStyle={styles.listItemLabelStyle}
                        ArrowDownIconComponent={() => <ArrowDownIcon/>}
                        ArrowUpIconComponent={() => <ArrowUpIcon/>}
                        TickIconComponent={() => <TickIcon/>}
                    />

                    {/*ORIGIN FILTER DROPDOWN*/}
                    <DropDownPicker
                        placeholder={"Origin"}
                        open={nationalityOpen}
                        value={nationalityValue}
                        items={nationalityItems}
                        setOpen={setNationalityOpen}
                        onOpen={onNationalityOpen}
                        setValue={setNationalityValue}
                        setItems={setNationalityItems}
                        zIndex={3000}
                        zIndexInverse={1000}
                        containerStyle={genreOpen ? {marginTop: 200} : styles.picker}
                        style={styles.dropdownInside}
                        labelStyle={styles.labelStyle}
                        placeholderStyle={styles.dropdownPlaceholder}
                        dropDownContainerStyle={styles.dropDownContainerStyle}
                        listItemLabelStyle={styles.listItemLabelStyle}
                        ArrowDownIconComponent={() => <ArrowDownIcon/>}
                        ArrowUpIconComponent={() => <ArrowUpIcon/>}
                        TickIconComponent={() => <TickIcon/>}
                    />
                </View>

                {/*CLEAR FILTERS BUTTON*/}
                <TouchableOpacity style={[styles.clearButton]} onPress={() => {
                    setStyleValue("")
                    setGenreValue("")
                    setNationalityValue("")
                    toggleModal();
                    clearFilter()
                }}>
                    <Text style={styles.buttonText}>Clear filters</Text>
                </TouchableOpacity>

                {/*APPLY FILTERS BUTTON*/}
                <TouchableOpacity style={[styles.button]} onPress={() => {
                    toggleModal();
                    toggleFilter()
                }}>
                    <Text style={styles.buttonText}>Apply filters</Text>
                </TouchableOpacity>
            </View>
        </Modal>
    )
}
