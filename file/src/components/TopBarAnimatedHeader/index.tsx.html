    import React from 'react';
import {Animated, TouchableOpacity} from 'react-native';
import styles from "./styles";
import {AntDesign, FontAwesome} from "@expo/vector-icons";
import {HeaderType, TopBarAnimatedHeaderProps} from "../../../types";
import {useInteractions} from "../../contexts/UserInteractions";

// CONSTANTS
const HEADER_MAX_HEIGHT = 250;
const HEADER_MIN_HEIGHT = 80;
const HEADER_SCROLL_DISTANCE = HEADER_MAX_HEIGHT - HEADER_MIN_HEIGHT;
const SUBSTRACT_START_TRANSITION = 30;
const ICON_TRANSITION = 20;

/**
 * Component used to display an animated TopBar on scroll down
 *
 * <img src="https://drive.google.com/uc?export=view&id=1ZHKJYfjeO0PhRRSpBXKtNwTTyRnsSl-3" alt="ArtworkPage" height="50">
 * @param {TopBarAnimatedHeaderProps} props
 * @param {string} props.title - Title of the top bar
 * @param {HeaderType} props.type - Type of header
 * @param {Animated.value} props.scrollY - Value of the scroll position
 * @param {string} [props.idArtwork] - ID of the artwork
 * @constructor
 * @return TopBarAnimatedHeader
 */
export default function TopBarAnimatedHeader(props: TopBarAnimatedHeaderProps) {

    const userInt = useInteractions()

    //TITLE MOVING UP WHEN SCROLLING DOWN
    const titleTranslateY = props.scrollY.interpolate({
        inputRange: [0, HEADER_SCROLL_DISTANCE - SUBSTRACT_START_TRANSITION, HEADER_SCROLL_DISTANCE],
        outputRange: [0, 10, 0],
        extrapolate: 'clamp',
    });

    //TITLE OPACITY GOING UP WHEN SCROLLING DOWN
    const titleOpacity = props.scrollY.interpolate({
        inputRange: [0, HEADER_SCROLL_DISTANCE - SUBSTRACT_START_TRANSITION, HEADER_SCROLL_DISTANCE],
        outputRange: [0, 0, 1],
        extrapolate: 'clamp',
    });

    //ENTIRE TOP BAR OPACITY GOING UP WHEN SCROLLING DOWN
    const topBarOpacity = props.scrollY.interpolate({
        inputRange: [0, HEADER_SCROLL_DISTANCE - ICON_TRANSITION, HEADER_SCROLL_DISTANCE],
        outputRange: [0, 0, 1],
        extrapolate: 'clamp',
    });

    return (
        <Animated.View style={[styles.topBarContainer, {opacity: topBarOpacity}]}>
            <Animated.Text
                style={[styles.artworkNameTitle, {transform: [{translateY: titleTranslateY}]}, {opacity: titleOpacity}]}>{props.title}</Animated.Text>

            {props.type === HeaderType.Artwork && <Animated.View style={[styles.iconsContainer]}>
                <TouchableOpacity onPress={() => {
                    userInt.setSyncIsLiked(!userInt.syncIsLiked);
                    //@ts-ignore
                    userInt.addDeleteLike(props.idArtwork, userInt.syncIsLiked)
                }}>
                    <AntDesign name={userInt.syncIsLiked ? "heart" : "hearto"} style={styles.icon} size={30}
                               color={"white"}/>
                </TouchableOpacity>
                <FontAwesome name={"bookmark-o"} style={styles.icon} size={30} color={"white"}/>
            </Animated.View>}

        </Animated.View>
    )
}
