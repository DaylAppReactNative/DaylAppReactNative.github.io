    import React, {useEffect, useState} from 'react';
import axios from "axios";
import {URL} from "../../../services/constantes";
import {useNavigation} from "@react-navigation/native";
import {useAuth} from "../../../contexts/Auth";
import {Text, TouchableOpacity, View} from "react-native";
import styles from "../../DiscoverByColor/styles";
import {Entypo} from "@expo/vector-icons";
import ArtworkCarouselSingle from "../../ArtworkCarouselSingle";

/**
 * Component displaying an {@link ArtworkCarouselSingle} of liked artworks
 *
 * <img src="https://drive.google.com/uc?export=view&id=17_yTpjYuS72FhFQ6SZGlj8Opw5MBNRSe" alt="ArtworkPage" height="500">
 * @constructor
 * @return LikedArtworksCarousel
 */
const LikedArtworksCarousel = () => {

    const navigation = useNavigation()
    const auth = useAuth()

    const [painting, setPainting] = useState<any>([]);

    // API CALL TO GET ALL LIKED ARTWORKS OF USER AND PARSE THEM INTO `ARTWORK`
    useEffect(() => {
        setPainting([])
        auth.authData.paintingLiked && auth.authData.paintingLiked.forEach((paintingId) => {
            axios.get(URL + `paintingEntity/${paintingId}`).then((response) => {
                let data = response.data;
                let parsedData = {
                    id: data.id,
                    imageUri: URL + `images/content/mini/${data.artist_idtf_intern}/${data.generatedId}`,
                    artistName: data.artist_name_tot,
                    artworkName: data.aw_title_Eng,
                    artistId: data.artist_idtf_intern
                }
                setPainting((painting: any) => [...painting, parsedData])
            });
        })
    }, [auth.authData.paintingLiked, auth.refreshKey])

    // useEffect(() => {
    //     setPainting([])
    //     if (auth.authData.paintingLiked) {
    //         for (let i = 0; i < auth.authData.paintingLiked.length; i++)
    //             axios.get(URL + `paintingEntity/${auth.authData.paintingLiked[i]}`).then((response) => {
    //                 let data = response.data;
    //                 let parsedData = {
    //                     id: data.id,
    //                     imageUri: URL + `images/content/mini/${data.artist_idtf_intern}/${data.generatedId}`,
    //                     artistName: data.artist_name_tot,
    //                     artworkName: data.aw_title_Eng,
    //                     artistId: data.artist_idtf_intern
    //                 }
    //
    //                 setPainting(painting => [...painting, parsedData]);
    //             });
    //     }
    // }, [auth.authData.paintingLiked, auth.refreshKey]);

    //IF THE USER HAS LIKED SOME ARTWORKS THEN SHOW IN CAROUSEL ELSE SHOW
    return (
        <View style={styles.container}>

            {/*IF CLICK ON TITLE THEN NAVIGATE TO DETAILED PAGE FULL OF ARTWORKS*/}
            <TouchableOpacity onPress={() => navigation.navigate("AllMyArtworksPage")}>
                <View style={styles.textAndChevron}>
                    {/*TEXT*/}
                    <Text style={styles.title}>Liked artworks</Text>
                    {/*CHEVRON*/}
                    <Entypo name={"chevron-right"} size={30} color={"white"}/>
                </View>
            </TouchableOpacity>

            {/*CAROUSEL OF RANDOM COLOR PAINTING*/}
            <ArtworkCarouselSingle data={painting}/>
        </View>
    )

    //return !auth.authData.paintingLiked || auth.authData.paintingLiked.length === 0  ? <View><Text style={styles.noLikedArtworks}>No liked artworks </Text></View> : <ArtworkCarouselDouble title={"Liked artworks"} data={painting} navigate={() => navigation.navigate("AllMyArtworksPage")}/>
}

export default LikedArtworksCarousel;
