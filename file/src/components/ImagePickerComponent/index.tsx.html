    import {Dimensions, StyleSheet, Text, TouchableOpacity, View} from 'react-native';
import * as ImagePicker from 'expo-image-picker';
import React from "react";
import Modal from "react-native-modal";

// PROPS
export type ImagePickerProps = {
    isImagePickerVisible: boolean,
    setImagePickerVisible(isImagePickerVisible: boolean): void,
    result: any,
    setResult(result: any): void,
    callback(): void,
}

/**
 * Component used to upload a picture of the user (from his files or his camera)
 *
 * <img src="https://drive.google.com/uc?export=view&id=13AMfMb0I_wcBY2S0SN77OTnvsTkzsphu" alt="ArtworkPage" height="500">
 * @param {ImagePickerProps} props
 * @param {boolean} props.isImagePickerVisible - Visibility of ImagePickerModal
 * @param {void} props.setImagePickerVisible - Toggles visibility of ImagePickerModal
 * @param {any} props.result - Parsed object corresponding to the image picked
 * @param {void} props.setResult - Function setting the result
 * @param {void} props.callback - Function called when the upload is over
 * @constructor
 * @return Modal
 */
export default function ImagePickerComponent(props: ImagePickerProps) {

    // TOGGLE MODAL
    const toggleModal = () => {
        props.setImagePickerVisible(!props.isImagePickerVisible)
    }

    // LAUNCH CAMERA OR FILE MANAGER
    const pickImage = async (type: string) => {
        // CHECK PERMISSIONS
        const permissionResult = type == "camera" ? await ImagePicker.requestCameraPermissionsAsync() : await ImagePicker.requestMediaLibraryPermissionsAsync();
        if (!permissionResult.granted) {
            // @ts-ignore
            alert("You've refused to allow this app to access your photos!");
            return;
        }

        // CHOOSE PICTURE
        let result = type == "camera" ? await ImagePicker.launchCameraAsync({
            mediaTypes: ImagePicker.MediaTypeOptions.Images,
            allowsEditing: true,
            quality: 0,

        }) : await ImagePicker.launchImageLibraryAsync({
            mediaTypes: ImagePicker.MediaTypeOptions.Images,
            allowsEditing: true,
            quality: 0,
        });

        // CASE NO PICTURE WERE CHOSEN
        if (result.cancelled) {
            return
        }

        // CASE PICTURE IS VALID => SEND DATA TO API
        if (result) {

            props.setResult(result)
            props.setImagePickerVisible(false)
            props.callback()
        }
    }

    return (
        <Modal animationIn={"slideInUp"}
               deviceHeight={Dimensions.get("window").height + 100}
               deviceWidth={Dimensions.get("window").width}
               isVisible={props.isImagePickerVisible}
               backdropTransitionOutTiming={0}
               statusBarTranslucent={true}>
            <View style={styles.container}>
                {/*TITLE*/}
                <Text style={styles.title}>Choose an image</Text>

                {/*UPLOAD FROM CAMERA*/}
                <TouchableOpacity
                    onPress={() => pickImage("camera")} style={styles.button}>
                    <Text style={styles.buttonText}>Use Camera</Text>
                </TouchableOpacity>

                {/*UPLOAD FROM FILES*/}
                <TouchableOpacity onPress={() => pickImage("file")} style={styles.button}>
                    <Text style={styles.buttonText}>Select from files</Text>
                </TouchableOpacity>

                {/*CANCEL*/}
                <TouchableOpacity onPress={() => toggleModal()} style={styles.button}>
                    <Text style={styles.buttonText}>Cancel</Text>
                </TouchableOpacity>
            </View>
        </Modal>
    );
}

const screenWidth = Dimensions.get('screen').width
const screenHeight = Dimensions.get('screen').height
