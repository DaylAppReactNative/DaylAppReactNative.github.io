    src/components/SquareComponentGrid/ListGrid/index.tsx

    import React, {useEffect, useState} from 'react';
import {SafeAreaView, Text, TouchableOpacity, View} from 'react-native';
import {AntDesign, Entypo} from "@expo/vector-icons";
import {List, ListActions, SquareComponentType} from "../../../../types";
import SquareComponentGrid from "../index";
import styles from "../styles";
import {useNavigation} from "@react-navigation/native";
import SmallerSearchBar from "../../Utilities/SearchBar/SmallerSearchBar";
import Toast from "react-native-toast-message";
import NewListModal from "../../Modals/NewListModal";
import {useInteractions} from "../../../contexts/UserInteractions";

// PROPS
export type ListGridProps = {
    title: string,
    data: List[],
    action: ListActions,
}

/**
 * Component used to display a button + grid of lists
 *
 * <img src="https://drive.google.com/uc?export=view&id=1BIMK3rKYplHSR9utN4IDhqq7uoy_QBrS" alt="ArtworkPage" height="300">
 * @param {ListGridProps} props
 * @param {string} props.title - Title of the grid
 * @param {[List] | []} props.data - An array of {@link List}
 * @param {ListActions} [props.action] - Type of action
 * @constructor
 * @return ListGridComponent
 */
export default function ListGridComponent(props: ListGridProps) {

    const [modalNewList, setModalNewList] = useState(false)
    const [clicked, setClicked] = useState(false)
    const [searchPhrase, setSearchPhrase] = useState("")
    const [newData, setNewData] = useState<any>(props.data)

    const navigation = useNavigation()
    const userInt = useInteractions()

    useEffect(() => {
        setNewData(props.data)
    }, [])

    const onPressCreateNewList = () => {
        setModalNewList(true)
    }

    const onPressSubmitNewList = (listName: string) => {
        if (listName) {
            userInt.createNewList(listName);
            setModalNewList(false);
        } else {
            Toast.show({
                    type: "dayl",
                    props: {
                        title: "Failed",
                        text: "No name was given to the list !",
                    }
                }
            )
        }
    }

    useEffect(() => {
        const temp = props.data.filter(item => item.title.includes(searchPhrase))
        //@ts-ignore
        setNewData(temp)
    }, [searchPhrase])

    return (
        <SafeAreaView style={styles.container}>

            {/*NEW LIST MODAL*/}
            <NewListModal modalNewList={modalNewList} setModalNewList={setModalNewList}/>

            <View style={props.action === ListActions.detailed ? styles.DetailedHeaderListGrid : styles.HeaderListGrid}>
                {/*HEADER LISTS GRID*/}
                <TouchableOpacity onPress={() => navigation.navigate('AllMyListsPage')}>
                    <View style={styles.textAndChevron}>
                        {/*TEXT*/}
                        <Text
                            style={props.action === ListActions.detailed ? styles.detailedTitle : styles.title}>{props.title}</Text>
                        {/*CHEVRON*/}
                        {props.action === ListActions.view &&
                            <Entypo name={"chevron-right"} size={30} color={"white"}/>}

                    </View>
                </TouchableOpacity>

                {/* PLUS BUTTON*/}
                <View style={styles.plusButton}>
                    <TouchableOpacity style={styles.plusButton} onPress={() => onPressCreateNewList()}>
                        <AntDesign name={"plus"} size={props.action === ListActions.detailed ? 35 : 25}
                                   color={"white"}/>
                    </TouchableOpacity>
                </View>

            </View>
            {props.action === ListActions.detailed &&
                <SmallerSearchBar clicked={clicked} searchPhrase={searchPhrase} setSearchPhrase={setSearchPhrase}
                                  setClicked={setClicked}/>}

            <SquareComponentGrid title={props.title} data={searchPhrase === "" ? props.data : newData}
                                 type={SquareComponentType.list}
                                 action={props.action}/>
        </SafeAreaView>
    )
}
