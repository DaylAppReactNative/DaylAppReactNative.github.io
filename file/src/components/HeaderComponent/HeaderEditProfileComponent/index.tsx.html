    // API CALL TO POST NEW BANNER IMAGE
    const sendDataBackground = (result: any) => {

        const xhr = new XMLHttpRequest();

        xhr.open('POST', URL + `images/uploadUserBackgroundIcon?userId=${auth.authData.id}`);

        const formData = new FormData();

        formData.append('image', {
            // @ts-ignore
            uri: result.uri,
            type: result.type + "/" + result.uri.split("/")[11],
            name: result.uri.split("/")[11]
        });

        xhr.send(formData);

        xhr.onreadystatechange = function () {
            if (xhr.status === 200) {
                Toast.show({
                    type: "dayl",
                    props: {
                        title: "Success",
                        text: "Changed banner",
                    }
                })
                xhr.abort()
                return
            } else {
                Toast.show({
                    type: "dayl",
                    props: {
                        title: "Error",
                        text: "Could not change banner",
                    }
                })
                xhr.abort()
                return
            }
        }
    }

    // API CALL TO POST NEW ICON IMAGE
    const sendDataProfile = (result: any) => {

        const xhr = new XMLHttpRequest();

        xhr.open('POST', URL + `images/uploadUserIcon?userId=${auth.authData.id}`);

        const formData = new FormData();

        formData.append('image', {
            // @ts-ignore
            uri: result.uri,
            type: result.type + "/" + result.uri.split("/")[11],
            name: result.uri.split("/")[11]
        });

    const pickImageBackground = async () => {
        // CHOOSE PICTURE

            mediaTypes: ImagePicker.MediaTypeOptions.Images,
            allowsEditing: true,
            quality: 0,
        });

        if (result.cancelled) {
            return
        }

        // CASE PICTURE IS VALID => SEND DATA TO API
        if (result) {
            sendDataBackground(result)
        }
    }

    const pickImageProfile = async () => {
        // CHOOSE PICTURE
        await ImagePicker.launchImageLibraryAsync({
            mediaTypes: ImagePicker.MediaTypeOptions.Images,
            allowsEditing: true,
            quality: 0,
        }).then((result) => {
            // CASE NO PICTURE WERE CHOSEN
            if (result.cancelled) {
                return
            }

            // CASE PICTURE IS VALID => SEND DATA TO API

                sendDataProfile(result)
            }
        })
    }

                {/*LIST AND AT LEAST FOUR IMAGES */}
                <View>
                    <Animated.Image style={styles.coverImage}
                                    source={{uri: URL + `images/contentUser/${auth.authData.id}/${auth.authData.backgroundImage}`}}
                                    resizeMode={'cover'}/>
                </View>

                {/*EDIT ICON*/}

                    pickImageBackground();
                    auth.setRefreshKey(auth.refreshKey + 1)

                           resizeMode={"cover"}/>
