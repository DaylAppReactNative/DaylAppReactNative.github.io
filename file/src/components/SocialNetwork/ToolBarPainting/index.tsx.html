import {TouchableOpacity, View} from "react-native";

import {styles} from "./styles";
import PostPainting from "../Posts/PostPainting";

/**
 * Component used to display a toolbar for {@link PostPainting}
 *
 * <img src="https://drive.google.com/uc?export=view&id=1247pwDo_OdM8UrJRodzvu4ILyXyE5nJw" alt="ArtworkPage" height="50">
 * @constructor
 * @return ToolBarPainting
 */
export default function ToolBarPainting() {
    return (
        <View style={styles.iconContainer}>
            {/*LIKE ICON*/}
            <TouchableOpacity style={styles.icon} onPress={() => console.log("Like")}>
                <AntDesign name={"heart"} size={20} color={"white"}/>
            </TouchableOpacity>
            {/*SAVE ICON*/}
            <TouchableOpacity style={styles.icon} onPress={() => console.log("Save")}>
                <FontAwesome name={"bookmark"} size={20} color={"white"}/>
            </TouchableOpacity>
            {/*SHARE ICON*/}
            <TouchableOpacity style={styles.icon} onPress={() => console.log("Share")}>
                <Feather name={"share"} size={20} color={"white"}/>
            </TouchableOpacity>
        </View>
    )
}
